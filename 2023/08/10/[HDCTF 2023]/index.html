小山 Hill Cipher 希尔密码 发明者为Lester S. Hill

# 希尔密码
# http://www.practicalcryptography.com/ciphers/hill-cipher/ 

c = 'eightfourtwozeroeightfourtwoonezeroeighteightfourzerotwofourzeroeightfourzeroonezeroonetwofour'
if 'eight' in c:
    c = c.replace('eight','8')
if 'four' in c:
    c = c.replace('four','4')
if 'zero' in c:
    c = c.replace('zero','0')
if 'one' in c:
    c = c.replace('one','1')
if 'two' in c:
    c = c.replace('two', '2')
print(c)

y = c.split('0')
s = ''
for i in range(len(y)):
    ch = ''
    count = 0
    for j in range(len(y[i])):
        ch = y[i][j]
        count += int(ch)
    s += chr(count+64)
print(s)
