from Crypto.Util.number import *
from pwn import *
import gmpy2
context.log_level = 'debug'
p = 2 ** 1024 - 2 ** 234 - 2 ** 267 - 2 ** 291 - 2 ** 403 - 1
def get_root(x,p):
    # g^x==4 mod p
    gg,a,b=gmpy2.gcdext(x,p-1)
    if gg==2:
        t=pow(4,a,p)
        g=pow(t,(p+1)//4,p)
        return g
    elif gg==1:
        return pow(4,a,p)
s='1'
r=remote('47.95.3.91', 42259)
while True:
    r.sendline('t')
    g=get_root(int(s,2),p)
    r.sendline(str(g))
    r.recvuntil(' desired integer: \n')
    tr=r.recvline()
    if b'flag' in tr:
        print(tr)
        break
    tt,rr=eval(tr[8:])
    if tt==1:
        s=s+'1'
    else:
        if s[-1]=='1':
            s=s[:-1]+'0'
    print(s)
r.interactive()
